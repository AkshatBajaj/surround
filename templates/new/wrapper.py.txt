from surround import Wrapper, AllowedTypes
from stages import ValidateData, PredictData, TrainData, {project_name}Data

class PipelineWrapper(Wrapper):
    def __init__(self):
        super().__init__(AllowedTypes.FILE)
        self.stages = [ValidateData()]

    def predict(self, input_data):
        self.stages.append(PredictData())
        self.data = OneData(input_data)

    def train(self, input_data):
        self.stages.append(TrainData())
        self.data = OneData(input_data)
